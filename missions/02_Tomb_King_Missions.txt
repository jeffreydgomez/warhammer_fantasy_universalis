		icon = mission_mosque
		required_missions = {}
		position = 6
		provinces_to_highlight = {
			OR = {
				AND = {
					NOT = { owned_by = ROOT }
					has_great_project = {
						type = tomb_of_settra
						tier = 0
					}
				}
				AND = {
					region = nehekhara_region
					NOT = {
						culture_group = ROOT
					}
				}
			}
		}
		trigger = {

		# Performance
			hidden_trigger = {
				OR = {
					ai = no
					num_of_cities = 25
				}
			}


			NOT = {
				any_province = {
					region = nehekhara_region
					NOT = {
						culture_group = ROOT
					}
				}
			}
			any_owned_province = {
				has_great_project = {
					type = tomb_of_settra
					tier = 0
				}
			}
		}
		effect = {
			add_country_modifier = { 
    			name = nehekhara_unify_culture
    			duration = -1  
			}
		}
	}
