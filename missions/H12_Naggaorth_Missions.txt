#Handle Clar Karond
h12_handle_clar_karond = {
    icon = nission_conquer_50_development
    required_missions = { h12_fate_of_dreadlords }
    provinces_to_highlight = {}
    trigger = {
        has_disaster = naggaroth_dreadlords_secede
        OR = {
            AND = {
                has_country_flag = h12_fight_clar_karond
                naggaroth_4_area = {
                    controlled_by = ROOT
                }
                naggaroth_5_area = {
                    controlled_by = ROOT
                }
                naggaroth_6_area = {
                    controlled_by = ROOT
                }
                black_forest_1_area = {
                    controlled_by = ROOT
                }
            }
            has_country_flag = h12_surrender_clar_karond
        }
    }
    effect = {
        remove_estate_privilege = estate_nobles_clar_karond_dreadlords
        if = {
            limit = {
                has_country_flag = h12_fight_clar_karond
            }
            set_country_flag = defeated_clar_karond
            prestige = 20
            legitimacy = 30
        }
        if = {
            limit = {
                has_country_flag = h12_surrender_clar_karond
            }
            set_country_flag = clar_karond_secedes
            prestige = -10
            legitimacy = -10
            naggaroth_4_area = {
                every_owned_province = {
                    cede_province = H17
                    clear_rebels = yes 
                }
            }
            naggaroth_5_area = {
                every_owned_province = {
                    cede_province = H17
                    clear_rebels = yes 
                }
            }
            naggaroth_6_area = {
                every_owned_province = {
                    cede_province = H17
                    clear_rebels = yes 
                }
            }
            black_forest_1_area = {
                every_owned_province = {
                    cede_province = H17
                    clear_rebels = yes 
                }
            }
        }
    }
}
}
