# Triggers and effects will be given the Actor's scope, with the Recipient's as FROM

# Create new diplomatic actions like this. AI will currently never send them.

annex_subjugated_warband =  {
	category = influence
	require_acceptance = no
	is_visible = {
		ai = no
		overlord_of = FROM
		num_of_subjects = 1
		OR = {
			not = {
				has_reform = chaos_warband
			}
		}
		FROM = {
			is_subject_of_type = chaos_warband_vassal
		}
	}
	
	is_allowed = {
		hidden_trigger = {
			ai = no
			overlord_of = FROM
			num_of_subjects = 1
			OR = {
				not = {
					has_reform = chaos_warband
				}
			}
			FROM = {
				is_subject_of_type = chaos_warband_vassal
			}
		}
		FROM = { NOT = { num_of_cities = 2 } }
		adm_power = 50
		is_at_war = no
	}
	
	on_accept = {
		add_adm_power = -50
		#Boost Capital Dev
		if = {
			limit = { capital_scope = { NOT = { development = 30 } } }
			capital_scope = {
				random_list = {
					50 = {
						add_base_tax = 1
						add_base_manpower = 1
					}
					50 = {
						add_base_manpower = 1
						add_base_production = 1
					}
				}
			}
		}
		
		#Annex FROM
		hidden_effect = {
			FROM = {
				if = {
					limit = { is_neighbor_of = ROOT }
					every_owned_province = {
						remove_core = FROM
						add_core = ROOT
						cede_province = ROOT
					}
					ROOT = {
						chaos_colonization = yes
						chaos_colonization = yes
					}
				}
				else = {
					every_owned_province = {
						decolonize = yes
						remove_core = PREV
					}
					ROOT = {
						chaos_colonization = yes
						chaos_colonization = yes
						chaos_colonization = yes
					}
				}
			}
		}
		
	}
	
	ai_will_do = {
		always = yes
	}
}
