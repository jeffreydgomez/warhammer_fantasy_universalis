# INSTRUCTIONS:
#
# condition				A diplomatic action can have any number of condition blocks, each with its own
#						tooltip, potential and allow section
#
# 	tooltip					Sets a custom text string similar to the hardcoded limits
# 							If no tooltip is scripted, the tooltip for the actual trigger will be shown
#							Note that the custom tooltip is only shown if the allow trigger is NOT met
#
# 	potential				Determines if the trigger is applicable or not
# 	allow					Determines if the action is valid or not
#
# effect				A diplomatic action can only have one effect block

# ROOT					actor
# FROM					target

### Phoenix Crown

dip_request_phoenix_crown = {

	category = trust

	require_acceptance = yes

	is_visible = {
		FROM = { any_owned_province = { has_province_modifier = location_of_the_phoenix_crown } }
	}

	is_allowed = {
		hidden_trigger = {
			OR = {
				ai = no
				AND = {
					FROM = { trust = { who = ROOT value = 100 } }
					FROM = { owes_favors = { who = ROOT value = 100 } }
				}
			}
		}

		OR = {
			FROM = { is_at_war = no }
			capital_scope = { infantry_in_province = FROM }
			capital_scope = { cavalry_in_province = FROM }
			FROM = { capital_scope = { infantry_in_province = ROOT } }
			FROM = { capital_scope = { cavalry_in_province = ROOT } }
		}

	}

    on_accept = {

    	add_favors = {
		   who = FROM
		   amount = -100
		}

		every_province ={
			limit = { has_province_modifier = location_of_the_phoenix_crown }
			remove_province_modifier = location_of_the_phoenix_crown
		}
		capital_scope = {
			add_permanent_province_modifier = { 
			    name = location_of_the_phoenix_crown
			    duration = -1  
			}
		}

    }

    ai_acceptance = {
		add_entry = {
			name = AI_TRUST_TOWARDS_ROOT
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}			
			subtract_variable = { 
				which = ai_value
				value = 99
			}
		}

		add_entry = {
            name = DIPLO_OPPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			subtract_variable = { 
				which = ai_value
				value = 199
			}
        }

    }

    ai_will_do = {
		always = no
    }
}
