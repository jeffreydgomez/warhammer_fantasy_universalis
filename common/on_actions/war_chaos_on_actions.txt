## KHORNE

on_battle_won_country = {
    if = {
        limit = {
            ROOT = { religion = khornate_religion }
        }
        ROOT =  {
            add_church_power = 5
        }
    }
}

on_siege_won_country = {
    if = {
        limit = {
            ROOT = { religion = khornate_religion }
        }
        ROOT =  {
            add_church_power = 2
        }
    }
}

on_main_war_won = {
    if = {
        limit = {
            ROOT = { religion = khornate_religion }
        }
        ROOT = { 
            country_event = { id = khorne_events.1 }
        }
    }
}

## SLAANESH

on_create_vassal = {
    if = {
        limit = {
            FROM = { religion = slaaneshi_religion }
        }
        FROM = { 
            country_event = { id = slaanesh_events.1 }
        }
    }
}

on_alliance_created = {
    if = {
        limit = {
            ROOT = {
                religion = slaaneshi_religion
            }
        }
        ROOT = { add_church_power = 10 }
    }
    if = {
        limit = {
            FROM = {
                religion = slaaneshi_religion
            }
        }
        FROM = { add_church_power = 10 }
    }
}

on_royal_marriage  = {
    if = {
        limit = {
            ROOT = {
                religion = slaaneshi_religion
            }
        }
        ROOT = { add_church_power = 10 }
    }
    if = {
        limit = {
            FROM = {
                religion = slaaneshi_religion
            }
        }
        FROM = { add_church_power = 10 }
    }
}
