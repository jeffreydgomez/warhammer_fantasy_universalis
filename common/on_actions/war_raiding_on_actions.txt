### Skaven have raiding, humans don't. If humans take a province, then skaven take it back, it shouldn't be raided. If Skaven take it back, then they are both the Owner and Controller.

### Raiding # FROM = Loser country?
on_siege_won_province = {
	if = {
		limit = {
			controller = {
				has_country_modifier = lets_raid_modifier
			}

			NOT = { has_province_modifier = recently_raided }
			NOT = {
				owner = {
					ROOT = {
						controlled_by = PREV
					}
				}
			}
		}

		if = {
			limit = {
				base_tax = 2
			}
			add_base_tax = -1
			controller = { add_treasury = 33 }
					
			add_permanent_province_modifier = { 
			    name = recently_raided
			    duration = 3650  
			}
			add_devastation = 10
			owner = { add_war_exhaustion = 0.05 }
		}

		if = {
			limit = {
				base_production = 2
			}
			add_base_production = -1
			controller = { add_treasury = 33 }

			add_permanent_province_modifier = { 
			    name = recently_raided
			    duration = 3650  
				}
			add_devastation = 10	
			owner = { add_war_exhaustion = 0.05 }
		} 

		if = {
			limit = {
				base_manpower = 2
			}
			add_base_manpower = -1
			controller = { add_treasury = 33 }

			add_permanent_province_modifier = { 
			    name = recently_raided
			    duration = 3650  
			}
			add_devastation = 10
			owner = { add_war_exhaustion = 0.05 }
		}
	}
}

on_province_owner_change = {

	if = {
		has_province_modifier = recently_raided
	}
	add_nationalism = 10
}
