### A Challenger Approaches

country_event = { 
	id = greenskin_events.1
	title = greenskin_events.1.t 
	desc = greenskin_events.1.d 
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		OR = {
			culture_group = greenskins_group
			culture_group = hobgoblin_group
		}
	}

	mean_time_to_happen = {
		days = 1
	}

# I'm still the boss

	option = { 
		ai_chance = { factor = 50 }
		name = greenskin_events.1.a 

		add_legitimacy = -10

		add_piety = -0.1

		}

# The throne is theirs
	option = { 
		ai_chance = { factor = 50 }
		name = greenskin_events.1.b

		kill_ruler = yes

		add_piety = 0.1

	}

# Let's fight it out!
	option = { 
		ai_chance = { factor = 50 }
		name = greenskin_events.1.c


		random_list = {
			50 = {
				if = {
					limit = {
						mil = 6
					}
					add_mil_power = 100
				}
				else = {
					change_mil = 1
				}
			}

			50 = {
				kill_ruler = yes
			}
		}

		add_piety = 0.1
		add_prestige = 10
		add_legitimacy = 5

	}
}

### Loot Own Lands

	title = greenskin_events.3.t 
	desc = greenskin_events.3.d 
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		OR = {
			culture_group = greenskins_group

		}

		any_owned_province = {
			base_tax = 2
			base_production = 2
			base_manpower = 2
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = { 
		ai_chance = { factor = 5 }
		name = greenskin_events.3.a

		add_treasury = -100
		add_piety = -0.1

	}

	option = { 
		ai_chance = { factor = 15 }
		name = greenskin_events.3.b

		random_owned_province = {
			limit = {
				base_tax = 2
				base_production = 2
				base_manpower = 2
			}
			add_base_tax = -1
			add_base_production = -1
			add_base_manpower = -1
		}

		add_piety = 0.1
	}

	option = { 
		ai_chance = { factor = 1 }
		name = greenskin_events.3.c
		add_stability = -1

		add_piety = -0.1
	}
}

### Raiding Party

country_event = { 
	id = greenskin_events.6
	title = greenskin_events.6.t 
	desc = greenskin_events.6.d 
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		OR = {
			culture_group = greenskins_group
			culture_group = hobgoblin_group
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = { 
		ai_chance = { factor = 100 }
		name = greenskin_events.6.a
		add_piety = -0.1
	}

	option = { 
		ai_chance = { factor = 5 }
		name = greenskin_events.6.b
		add_piety = 0.1

		random_province = {
			limit = {
				is_neighbor_of = ROOT
				NOT = { owned_by = ROOT }
			}

			spawn_rebels = {
	  	 		size = 1
	    		type = noble_rebels
	    		culture = ROOT
	    		religion = ROOT
			}

			owner = {
				add_opinion = {
			    	who = ROOT
			    	modifier = ravaged_border
				}
			}

		}
	}
}
