# Our nobles clambor to reclaim Ulthuan and take the Phoenix throne! We must establish a beachhead.

country_event = { 
	id = dark_elf_events.1
	title = dark_elf_events.1.t 
	desc = dark_elf_events.1.d 
	picture = DARKELF_eventPicture

	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		culture_group = druchii_group
		
		is_great_power = yes
		is_at_war = no

		manpower_percentage = 0.50
		navy_size_percentage = 0.50
		army_size_percentage = 0.50

		NOT = {
			capital_scope = { superregion = ulthuan_superregion }
		}

		any_country = {
			capital_scope = { superregion = ulthuan_superregion }
			NOT = { truce_with = ROOT }
			ROOT = {
				NOT = {
					has_opinion = {
						who = PREV
						value = 0
					}
				}
				
				military_strength = {
					who = PREV
					value = 1.5
				}	
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {

		random_country = {
			limit = {
				capital_scope = { superregion = ulthuan_superregion }
				NOT = { truce_with = ROOT }
				ROOT = {
					NOT = {
						has_opinion = {
							who = PREV
							value = 0
						}
					}

					military_strength = {
						who = PREV
						value = 1.5
					}	

				}
			}
			capital_scope = { save_event_target_as = event_target }
		}
	}

	goto = event_target

# It is time for war!

	option = { 
		ai_chance = { factor = 50 }
		name = dark_elf_events.1.option_1 


		add_yearly_manpower = 0.5

		add_country_modifier = { 
		    name = war_better_morale
		    duration = 3650
		}

		add_country_modifier = { 
		    name = war_better_navy_morale
		    duration = 3650
		}

		event_target:event_target = {
			owner = {
				ROOT = {
					declare_war_with_cb = {
					    who = PREV
					    casus_belli = cb_annex
					}
				}
			}
		}
	}

# Now is not the time
	option = { 
		ai_chance = { factor = 0 }
		name = dark_elf_events.1.option_2


		add_prestige = -10

		add_estate_loyalty = {
		    estate = estate_nobles
		    loyalty = -10
		}	


	}
}

	}

# Let's fight it out!
	option = { 
		ai_chance = { factor = 0 }

			50 = {
				if = {
					limit = {

					}

					change_mil = 1
				}
			}

			50 = {
				kill_ruler = yes
			}
		}
		
		add_prestige = 10
		add_legitimacy = 5

	}
}

		}

		    name = war_witch_hunts

		}
	}

		ai_chance = { factor = 0 }
		name = dark_elf_events.4.c

		}
	}

## Slave Rebellion

country_event = { 
	id = dark_elf_events.7
	title = dark_elf_events.7.t 
	desc = dark_elf_events.7.d 
	picture = DARKELF_eventPicture

	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		OR = {
			culture_group = druchii_group
			religion = cytharai
		}
		any_owned_province = {
			trade_goods = slaves
			base_production = 3
		}

	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		random_owned_province = {
			limit = {
				trade_goods = slaves
			}
			save_event_target_as = event_target
		}
	}

	goto = event_target

# Break them Utterly

	option = { 
		ai_chance = { factor = 50 }
		name = dark_elf_events.7.option_1

		event_target:event_target = {
			add_base_production = -2
		}

	}

# Let them rise - it will be sport
	option = { 
		ai_chance = { factor = 50 }
		name = dark_elf_events.7.option_2

		event_target:event_target = {
			spawn_rebels = {
			    size = 1
			    type = particularist_rebels
			}
		}

	}

# Let the unrest build
	option = { 
		ai_chance = { factor = 50 }
		name = dark_elf_events.7.option_3

		add_stability = -1
	}
}
