	is_triggered_only = yes

	}

# I'm still the boss

	option = { 
		ai_chance = { factor = 50 }
		name = skaven.2.a 

		add_legitimacy = -10

		}

# The throne is theirs
	option = { 
		ai_chance = { factor = 50 }
		name = skaven.2.b

		kill_ruler = yes

	}

# Let's fight it out!
	option = { 
		ai_chance = { factor = 50 }
		name = skaven.2.c

			50 = {
				if = {
					limit = {

					}

				}
				else = {
					change_mil = 1
				}
			}

			50 = {
				kill_ruler = yes
			}
		}
		
		add_prestige = 10
		add_legitimacy = 5

	}
}



### Raiding Party. A group of our soldiers are getting uppity, and want to raid a neighbor. Should we let them?

country_event = { 
	id = skaven.3
	title = skaven.3.t 
	desc = skaven.3.d 
	picture = SKAVEN_eventPicture
	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		culture_group = skaven_group
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		random_province = {
			limit = {
				is_neighbor_of = ROOT
				NOT = { owned_by = ROOT }
			}
			save_event_target_as = event_target
		}
	}

	goto = event_target

# Better Not
	option = { 
		ai_chance = { factor = 100 }
		name = skaven.3.a
	}

## Send them out.
	option = { 
		ai_chance = { factor = 5 }
		name = skaven.3.b
		add_piety = 0.1

		event_target:event_target = {

			spawn_rebels = {
	  	 		size = 1
	    		type = raiding_party_rebels
	    		culture = ROOT
	    		religion = ROOT
			}

			owner = {
				add_opinion = {
			    	who = ROOT
			    	modifier = ravaged_border
				}
			}

		}
	}
}

### A new Warpstone deposit has been discovered! But it's under a thriving warren, and the only way to get at it is to destroy the warren.

country_event = { 
	id = skaven.5
	title = skaven.5.t 
	desc = skaven.5.d 
	picture = SKAVENTECH_eventPicture

	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		culture_group = skaven_group
		any_owned_province = {
			development = 9
			NOT = { trade_goods = warpstone }
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		random_owned_province = {
			limit = {
				development = 9
				NOT = { trade_goods = warpstone }
			}
			save_event_target_as = event_target
		}
	}

	goto = event_target

# Start blasting!

	option = { 
		ai_chance = { factor = 50 }
		name = skaven.5.a 

		event_target:event_target = {

			set_base_tax = 1
			set_base_production = 3
			set_base_manpower = 1

			change_trade_goods = warpstone
		}
	}

# Wait, there's an even bigger deposit here! I think...
	option = { 
		ai_chance = { factor = 50 }
		name = skaven.5.b

		event_target:event_target = {

			set_base_tax = 1
			set_base_production = 1
			set_base_manpower = 1

			random = { 
    			chance = 50 
    			change_trade_goods = warpstone
    			set_base_production = 10
			}
		}
	}

# Leave it alone ...
	option = { 
		ai_chance = { factor = 50 }
		name = skaven.5.c
		add_prestige = -5
	}
}

### Warpstone Addition - Skaven keep consuming the warpston

country_event = { 
	id = skaven.7
	title = skaven.7.t 
	desc = skaven.7.d 
	picture = SKAVENTECH_eventPicture

	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		culture_group = skaven_group
		any_owned_province = {
			trade_goods = warpstone
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		limit = {
			trade_goods = warpstone
			base_production = 2
		}
		random_owned_province = {
			save_event_target_as = event_target
		}
	}

	goto = event_target

# Grind it up and snort it!

	option = { 
		ai_chance = { factor = 50 }
		name = skaven.7.a 
		event_target:event_target = {
			add_base_production = -1
		}
		change_innovativeness = 2
	}

# Stop eating the warpstone!
	option = { 
		ai_chance = { factor = 50 }
		name = skaven.7.b

		event_target:event_target = {
			add_base_production = 3
		}

		add_stability = -1

	}

# Eh, let them do what they want
	option = { 
		ai_chance = { factor = 50 }
		name = skaven.7.c
	}
}
