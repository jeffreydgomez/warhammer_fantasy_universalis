##  A group of #### from our neighboring country want to come in.
country_event = { 
	id = neighbor_events.X
	title = neighbor_events.X.t 
	desc = neighbor_events.X.d 
	picture = COURT_eventPicture
	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		NOT = {
			culture_group = cathayan_cg
			religion = yin_yang
		}
		any_neighbor_country = {
			culture_group = cathayan_cg
			religion = yin_yang
			NOT = { war_with = ROOT }
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {

		random_neighbor_country = {
			limit = {
				culture_group = cathayan_cg
				religion = yin_yang
			}
			save_event_target_as = neighbor_country
		}

		event_target:neighbor_country = {
			capital_scope = { save_event_target_as = neighbor_capital }

		}

		random_owned_province = {
			limit = { 
				any_neighbor_province = { 
					owned_by = event_target:neighbor_country
				}
			}
			save_event_target_as = target_province
		}
	}

	goto = neighbor_capital

# Send them Away
	option = { 
		ai_chance = { factor = 50 }
		name = neighbor_events.X.option_1

		add_prestige = 5

	}

# Take them in
	option = { 
		ai_chance = { factor = 50 }
		name = neighbor_events.X.option_2
		goto = target_province

		event_target:target_province = {
			change_religion = yin_yang
			change_culture = peasant_cathayan

			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
		}
	}
}



##  Having Tileans as neighbors is a mixed blessing. A group of merchants from our neighboring country have tricked our own merchants.
country_event = { 
	id = neighbor_events.1
	title = neighbor_events.1.t 
	desc = neighbor_events.1.d 
	picture = TILEA_eventPicture
	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		NOT = {
			culture_group = tilean_group
			religion = myrmidia
		}
		any_neighbor_country = {
			culture_group = tilean_group
			religion = myrmidia
			NOT = { war_with = ROOT }
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {

		random_neighbor_country = {
			limit = {
				culture_group = tilean_group
				religion = myrmidia
			}
			save_event_target_as = neighbor_country
		}

		event_target:neighbor_country = {
			capital_scope = { save_event_target_as = neighbor_capital }

		}

		random_owned_province = {
			limit = { 
				any_neighbor_province = { 
					owned_by = event_target:neighbor_country
				}
			}
			save_event_target_as = target_province
		}
	}

	goto = neighbor_capital

# Capture the merchants and take their things by force!
	option = { 
		ai_chance = { factor = 50 }
		name = neighbor_events.1.option_1

		add_years_of_income = 0.5

		reverse_add_opinion = {
		    who = event_target:neighbor_country
		    modifier = war_poor_relations
		}
	}

# Keep our money! But teach us how you did that.
	option = { 
		ai_chance = { factor = 50 }
		name = neighbor_events.1.option_2

		add_mercantilism = 2
	}

# We can always use skilled merchants such as yourselves. Why not join us?
	option = { 
		ai_chance = { factor = 50 }
		name = neighbor_events.1.option_3
		goto = target_province

		add_mercantilism = 3

		event_target:target_province = {
			change_religion = myrmidia
			change_culture = reman

			add_base_tax = 1
		}

		reverse_add_opinion = {
		    who = event_target:neighbor_country
		    modifier = war_poor_relations
		}
	}
}
