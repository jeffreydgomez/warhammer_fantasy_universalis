namespace = neighbor_events
normal_or_historical_nations = yes


##  A group of #### from our neighboring country want to come in.
### There are benefits to neighboring XXX

country_event = { 
	id = neighbor_events.X
	title = neighbor_events.X.t 
	desc = neighbor_events.X.d 
	picture = COURT_eventPicture
	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		NOT = { culture_group = cathayan_cg }
		any_neighbor_country = { culture_group = cathayan_cg }

		NOT = { religion = yin_yang }
		any_neighbor_country = { religion = yin_yang }
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {

		if = {
			any_owned_province = {
				limit = {
					any_neighbor_province = {
						culture_group = cathayan_cg
						religion = yin_yang
					}
				}

				random_owned_province = {
					limit = {
						any_neighbor_province = {
							culture_group = cathayan_cg
							religion = yin_yang
						}
					}
					save_event_target_as = event_target
				}
			}
		} else = {
			random_owned_province = {
				save_event_target_as = event_target
			}
		}
	}

	goto = event_target

# Send them Away
	option = { 
		ai_chance = { factor = 50 }
		name = neighbor_events.X.option_1

		add_prestige = 5

	}

# Take them in
	option = { 
		ai_chance = { factor = 50 }
		name = neighbor_events.X.option_2

		event_target:event_target = {
			change_religion = yin_yang
			change_culture = peasant_cathayan

			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1

		}
	}
}
