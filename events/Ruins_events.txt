country_event = { 
	id = ruins_events.X
	title = ruins_events.X.title
	desc = ruins_events.X.desc
	picture = east_slavic_CITY_DEVELOPMENT_eventPicture

	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		any_owned_province = {
			has_terrain = urban_terrain
			is_colony = yes			
		}
	}

	immediate = {
		random_owned_province = {
			limit = {
				has_terrain = urban_terrain
				is_colony = yes
			}
			save_event_target_as = event_target
		}
	}

	goto = event_target

	mean_time_to_happen = {
		days = 1
	}

# FIRST OPTION

	option = { 
		ai_chance = { factor = 50 }
		name = ruins_events.X.a 

	}

# SECOND OPTION
	option = { 
		ai_chance = { factor = 50 }
		name = ruins_events.X.b


	}

# THIRD OPTION
	option = { 
		ai_chance = { factor = 50 }
		name = ruins_events.X.c
	}
}


### Colonists want to take some warpstone home
country_event = { 
	id = ruins_events.1
	title = ruins_events.1.title
	desc = ruins_events.1.desc
	picture = east_slavic_CITY_DEVELOPMENT_eventPicture

	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		any_owned_province = {
			has_terrain = urban_terrain
			is_colony = yes
			base_production = 2		
		}
	}

	immediate = {
		random_owned_province = {
			limit = {
				has_terrain = urban_terrain
				is_colony = yes
				base_production = 2
			}
			save_event_target_as = event_target
		}
	}

	goto = event_target

	mean_time_to_happen = {
		days = 1
	}

# Let's take a piece now and bring it home good idea

	option = { 
		ai_chance = { factor = 50 }
		name = ruins_events.1.a

		add_treasury = 100
		event_target:event_target = {
			add_base_production = -1
			add_colonysize = -25
		}

	}

# Leave it!
	option = { 
		ai_chance = { factor = 50 }
		name = ruins_events.1.b


	}
}

### Some colonists have disappeared. Should we send others to find them?

country_event = { 
	id = ruins_events.5
	title = ruins_events.5.title
	desc = ruins_events.5.desc
	picture = east_slavic_CITY_DEVELOPMENT_eventPicture

	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		any_owned_province = {
			has_terrain = urban_terrain
			is_colony = yes			
		}
	}

	immediate = {
		random_owned_province = {
			limit = {
				has_terrain = urban_terrain
				is_colony = yes
			}
			save_event_target_as = event_target
		}
	}

	goto = event_target

	mean_time_to_happen = {
		days = 1
	}

# No, let's not take the risk

	option = { 
		ai_chance = { factor = 50 }
		name = ruins_events.5.a 

		event_target:event_target = {

			add_colonysize = -25

		}

	}

# Yes, mount a search party
	option = { 
		ai_chance = { factor = 50 }
		name = ruins_events.5.b

		event_target:event_target = {

			add_colonysize = -25

			random_list = {
			    50 = {
			        add_colonysize = -25
			    }
			    50 = {
			        add_colonysize = 25
			    }
			}
		}



	}

# Mount a search party, and send some troops and supplies!
	option = { 
		ai_chance = { factor = 50 }
		name = ruins_events.5.c

		event_target:event_target = {

			add_colonysize = -25
			add_treasury = -50

			random_list = {
			    25 = {
			        add_colonysize = -25
			        add_manpower = -2
			    }
			    75 = {
			        add_colonysize = 25
			    }
			}
		}
	}
}

### Some young folk from the capital are interested in settling in the ruins. Should we let them go?

country_event = { 
	id = ruins_events.7
	title = ruins_events.7.title
	desc = ruins_events.7.desc
	picture = east_slavic_CITY_DEVELOPMENT_eventPicture

	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		any_owned_province = {
			has_terrain = urban_terrain
			is_colony = yes			
		}
		capital_scope = {
			base_manpower = 2
			base_production = 2
			base_tax = 2
		}
	}

	immediate = {
		random_owned_province = {
			limit = {
				has_terrain = urban_terrain
				is_colony = yes
			}
			save_event_target_as = event_target
		}
	}

	goto = event_target

	mean_time_to_happen = {
		days = 1
	}

# Yes, let them go.

	option = { 
		ai_chance = { factor = 50 }
		name = ruins_events.7.a

		capital_scope = {
			add_base_tax = -1
			add_base_production = -1
			add_base_manpower = -1
		}

		event_target:event_target = {
			add_colonysize = 25
		}

	}

# Them, and their friends!
	option = { 
		ai_chance = { factor = 50 }
		name = ruins_events.7.b

		capital_scope = {
			add_base_tax = -2
			add_base_production = -2
			add_base_manpower = -2
		}

		event_target:event_target = {
			add_colonysize = 50
		}

	}

# No, make them stay
	option = { 
		ai_chance = { factor = 50 }
		name = ruins_events.7.c
	}
}
