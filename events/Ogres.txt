### That livestock looks awfully tasty!

country_event = { 
	id = ogre_events.1
	title = ogre_events.1.t 
	desc = ogre_events.1.d 
	picture = OGRE_eventPicture
	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		OR = {
			culture_group = ogre_group
			religion = the_great_maw_religion
		}
		any_owned_province = {
			OR = {
				trade_goods = livestock
				trade_goods = wool
			}
			base_production = 3
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		random_owned_province = {
			limit = {
				OR = {
					trade_goods = livestock
					trade_goods = wool
				}
				base_production = 3
			}
			save_event_target_as = event_target
		}
	}

	goto = event_target

# Don't eat the livestock!

	option = { 
		ai_chance = { factor = 50 }
		name = ogre_events.1.option_1 

	}

# Eat up to increase our bulk!
	option = { 
		ai_chance = { factor = 0 }
		name = ogre_events.1.option_2

		event_target:event_target = {
			add_base_production = -2
		}

		add_country_modifier = { 
		    name = war_increased_shock_damage
		    duration = 3650
		}


	}

# Into the pot to feed our gut magic!
	option = { 
		ai_chance = { factor = 0 }
		name = ogre_events.1.option_3

		event_target:event_target = {
			add_base_production = -2
		}

		change_innovativeness = 1
	}
}

### A Challenger wants the throne.

country_event = { 
	id = ogre_events.5
	title = ogre_events.5.t 
	desc = ogre_events.5.d 
	picture = OGRE_eventPicture
	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		culture_group = ogre_group
	}

	mean_time_to_happen = {
		days = 1
	}

# I'm still the boss

	option = { 
		ai_chance = { factor = 50 }
		name = ogre_events.5.a 

		add_legitimacy = -10

		}

# The throne is theirs
	option = { 
		ai_chance = { factor = 50 }
		name = ogre_events.5.b

		kill_ruler = yes

	}

# Let's fight it out!
	option = { 
		ai_chance = { factor = 50 }
		name = ogre_events.5.c

		random_list = {
			50 = {
				if = {
					limit = {
						mil = 6
					}
					add_mil_power = 100
				}
				else = {
					change_mil = 1
				}
			}

			50 = {
				kill_ruler = yes
			}
		}
		
		add_prestige = 10
		add_legitimacy = 5

	}
}



### Raiding Party. A group of our soldiers are getting uppity, and want to raid a neighbor. Should we let them?

country_event = { 
	id = ogre_events.6
	title = ogre_events.6.t 
	desc = ogre_events.6.d 
	picture = OGRE_eventPicture
	
	fire_only_once = no
	is_triggered_only = yes
		
	trigger = {
		culture_group = ogre_group
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		random_province = {
			limit = {
				is_neighbor_of = ROOT
				NOT = { owned_by = ROOT }
			}
			save_event_target_as = event_target
		}
	}

	goto = event_target

# Better Not
	option = { 
		ai_chance = { factor = 100 }
		name = ogre_events.6.option_1
	}

## Send them out.
	option = { 
		ai_chance = { factor = 5 }
		name = ogre_events.6.option_2

		event_target:event_target = {

			spawn_rebels = {
	  	 		size = 1
	    		type = raiding_party_rebels
	    		culture = ROOT
	    		religion = ROOT
			}

			owner = {
				add_opinion = {
			    	who = ROOT
			    	modifier = ravaged_border
				}
			}

		}
	}
}
